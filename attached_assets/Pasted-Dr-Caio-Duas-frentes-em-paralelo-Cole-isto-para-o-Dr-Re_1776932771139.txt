WITH calc AS (
  SELECT pr.id,
    COALESCE(
      NULLIF(fb.valor_max, 0),            -- via blend quando disponível
      NULLIF(pr.valor_formula_real, 0),   -- declarado pela farmácia
      NULLIF(pr.valor_formula_estimado, 0)

    CASE
      WHEN fb.valor_max > 0 THEN 'formula_blend_valor_max'
      WHEN pr.valor_formula_real > 0 THEN 'valor_formula_real'

  JOIN farmacias_parmavault fp ON fp.id = pr.farmacia_id
  LEFT JOIN formula_blend fb ON fb.id = pr.blend_id
  WHERE pr.comissao_estimada IS NULL OR pr.comissao_estimada = 0
)
UPDATE parmavault_receitas pr SET
  comissao_estimada = CASE
    WHEN c.base_calculo > 0
      THEN ROUND(c.base_calculo * (c.percentual_comissao / 100), 2)

  comissao_estimada_origem = c.origem,
  comissao_estimada_em = now()
