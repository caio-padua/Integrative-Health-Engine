// lib/contratos/validarEmissao.ts
import { db } from "@workspace/db";
import { sql } from "drizzle-orm";

export async function verificarContratoNaEmissao(
  unidadeId: number,
  farmaciaId: number,
  usuarioId: number,
): Promise<{ temContrato: boolean; avisar: boolean; mensagem: string }> {
  const r = await db.execute(sql`
    SELECT id, tipo_relacao FROM farmacias_unidades_contrato
    WHERE unidade_id = ${unidadeId}
      AND farmacia_id = ${farmaciaId}
      AND ativo = true
      AND (vigencia_fim IS NULL OR vigencia_fim >= CURRENT_DATE)
    LIMIT 1
  `);

  const temContrato = r.rows.length > 0;

  // Log de auditoria sempre
  await db.execute(sql`
    INSERT INTO audit_logs (entidade, entidade_id, acao, usuario_id, meta)
    VALUES ('emissao_contrato_check', ${farmaciaId}, ${temContrato ? 'contrato_ok' : 'sem_contrato'}, ${usuarioId},
      ${JSON.stringify({ unidade_id: unidadeId, farmacia_id: farmaciaId, tem_contrato: temContrato })}::jsonb)
  `);

  return {
    temContrato,
    avisar: !temContrato,
    mensagem: temContrato
      ? ""
      : "⚠️ Sem contrato vigente com esta farmácia. Confirma emissão assim mesmo?",
  };
}

// scripts/job_retroativo_comissao.ts
import { db } from "@workspace/db";
import { sql } from "drizzle-orm";

export async function jobRetroativoComissao(): Promise<{
  atualizados: number; sem_valor: number; ja_calculados: number;
}> {
  // Só toca onde ainda não foi calculado
  const result = await db.execute(sql`
    UPDATE parmavault_receitas pr
    SET
      comissao_estimada = CASE
        WHEN pr.valor_formula_real IS NOT NULL AND pr.valor_formula_real > 0
          THEN ROUND(pr.valor_formula_real * (fp.percentual_comissao / 100), 2)
        ELSE NULL
      END,
      comissao_origem = CASE
        WHEN pr.valor_formula_real IS NOT NULL AND pr.valor_formula_real > 0
          THEN 'job_retroativo_' || to_char(now(), 'YYYY-MM-DD')
        ELSE 'sem_valor_base'
      END,
      comissao_calculado_em = now()
    FROM farmacias_parmavault fp
    WHERE fp.id = pr.farmacia_id
      AND pr.comissao_estimada IS NULL
    RETURNING
      pr.id,
      pr.valor_formula_real,
      pr.comissao_estimada
  `);

  const atualizados = result.rows.filter((r: any) => r.comissao_estimada !== null).length;
  const sem_valor = result.rows.filter((r: any) => r.comissao_estimada === null).length;

  console.log(`[job_retroativo] atualizados=${atualizados} sem\_valor=${sem_valor}`);
  return { atualizados, sem_valor, ja_calculados: 0 };
}

// lib/relatorios/gerarPdfReconciliacao.ts
import PDFDocument from "pdfkit";
import { createHash } from "crypto";

const NAVY = "#020406";
const GOLD = "#C89B3C";

function iniciaisPaciente(nome: string): string {
  return nome
    .split(" ")
    .filter(Boolean)
    .map((n) => n[0].toUpperCase() + ".")
    .join(" ");
}

function calcProtocolo(farmaciaId: number, periodoInicio: string, periodoFim: string): string {
  return createHash("sha256")
    .update(`${farmaciaId}-${periodoInicio}-${periodoFim}-${Date.now()}`)
    .digest("hex")
    .slice(0, 8)
    .toUpperCase();
}

export function gerarPdfReconciliacao(dados: {
  farmacia: { id: number; nome: string; percentual_comissao: number };
  periodo: { inicio: string; fim: string };
  resumo: { previsto: number; declarado: number; recebido: number; gap: number; n_receitas: number };
  receitas: Array<{
    id: number; data: string; paciente_nome: string;
    valor_formula: number; comissao_devida: number;
    status_declarado: string; status_pago: string;
  }>;
  graficoPng?: Buffer; // chartjs-node-canvas renderizado antes
}): { stream: PDFKit.PDFDocument; protocolo: string } {
  const protocolo = calcProtocolo(dados.farmacia.id, dados.periodo.inicio, dados.periodo.fim);
  const doc = new PDFDocument({ size: "A4", margin: 40, bufferPages: true });

  // PÁGINA 1 — CAPA
  doc.rect(0, 0, doc.page.width, 100).fill(NAVY);
  doc.rect(0, 96, doc.page.width, 4).fill(GOLD);
  doc.fillColor(GOLD).font("Helvetica-Bold").fontSize(18)

  // PÁGINA 2 — RESUMO EXECUTIVO (nova página)
  doc.addPage();
  // ... resumo + gráfico PNG embutido

  // PÁGINA 3 — TABELA DETALHADA
  doc.addPage();
  dados.receitas.forEach((r) => {
    // linha da tabela com iniciaisPaciente(r.paciente_nome)
  });

  // RODAPÉ em todas as páginas

  for (let i = 0; i < pages.count; i++) {
