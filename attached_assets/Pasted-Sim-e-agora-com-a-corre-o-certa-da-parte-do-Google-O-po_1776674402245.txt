A lógica do nome é:

Gerador do catálogo

export function generateClinicEmailCatalog(clinicId: string, clinicSlug: string) {
  const rows = [];
  for (const cargo of CARGOS) {
    for (const modo of MODOS) {
      for (const hierarquia of HIERARQUIAS) {
        const email = `${cargo}.${modo}.${hierarquia}.${clinicSlug}@${DOMAIN}`;
        rows.push({
          clinicId,
          cargo,
          modo,
          hierarquia,
          email,
          googleTargetUser: BASE_USERS_BY_CARGO[cargo],
          status: 'available' as const,
          createdAt: new Date(),
          selectedAt: null,
          provisionedAt: null,
          disabledAt: null,
          lastError: null,
        });
      }
    }
  }
  return rows;
}

Criação automática do catálogo quando nasce a clínica

export async function createClinicEmailCatalog(clinicId: string, clinicSlug: string) {
  const catalog = generateClinicEmailCatalog(clinicId, clinicSlug);
  await db.clinicEmailIdentity.createMany({
    data: catalog,
    skipDuplicates: true,
  });
  return catalog;
}

Toggle

export async function toggleEmailSelection(identityId: string, currentStatus: EmailIdentityStatus) {
  if (currentStatus === 'available') {
    return db.clinicEmailIdentity.update({
      where: { id: identityId },
      data: {
        status: 'selected',
        selectedAt: new Date(),
      },
    });
  }
  if (currentStatus === 'selected') {
    return db.clinicEmailIdentity.update({
      where: { id: identityId },
      data: {
        status: 'available',
        selectedAt: null,
      },
    });
  }
  throw new Error('Status inválido para toggle');
}

Provisionamento real no Google Workspace

Aqui entra a parte importante para o Replit.

Você vai usar um service account com delegação no Google Workspace, ou OAuth de admin, para chamar o endpoint oficial users.aliases.insert do Admin SDK. O endpoint oficial para adicionar alias é esse do Directory API.  ￼

Dependências

npm install googleapis

Função de autenticação

import { google } from 'googleapis';
export function getGoogleDirectoryClient() {
  const auth = new google.auth.GoogleAuth({
    credentials: {
      client_email: process.env.GOOGLE_SERVICE_ACCOUNT_EMAIL,
      private_key: process.env.GOOGLE_PRIVATE_KEY?.replace(/\\n/g, '\n'),
    },
    scopes: ['https://www.googleapis.com/auth/admin.directory.user.alias'],
    clientOptions: {
      subject: process.env.GOOGLE_ADMIN_IMPERSONATION_EMAIL,
    },
  });
  return google.admin({
    version: 'directory_v1',
    auth,
  });
}

Criar alias

export async function createGoogleAlias(userKey: string, aliasEmail: string) {
  const admin = getGoogleDirectoryClient();
  await admin.users.aliases.insert({
    userKey,
    requestBody: {
      alias: aliasEmail,
    },
  });
}

Provisionar os selecionados

export async function provisionSelectedEmails(clinicId: string) {
  const selected = await db.clinicEmailIdentity.findMany({
    where: {
      clinicId,
      status: 'selected',
    },
  });
  let successCount = 0;
  for (const item of selected) {
    try {
      await db.clinicEmailIdentity.update({
        where: { id: item.id },
        data: { status: 'provisioning' },
      });
      await createGoogleAlias(item.googleTargetUser, item.email);
      await db.clinicEmailIdentity.update({
        where: { id: item.id },
        data: {
          status: 'provisioned',
          provisionedAt: new Date(),
          lastError: null,
        },
      });
      successCount++;
    } catch (error: any) {
      await db.clinicEmailIdentity.update({
        where: { id: item.id },
        data: {
          status: 'failed',
          lastError: String(error?.message || error),
        },
      });
    }
  }
  return successCount;
}

export async function disableEmail(identityId: string) {
  const item = await db.clinicEmailIdentity.findUniqueOrThrow({
    where: { id: identityId },
  });

  return db.clinicEmailIdentity.update({

      status: 'disabled',
      disabledAt: new Date(),

}
