  return Object.entries(STATUS_BANDS).find(
    ([, b]) => value >= b.range[0] && value <= b.range[1]

router.get('/global', requireMaster, async (req, res) => {
  const { rows } = await pool.query(`
    SELECT
      COUNT(DISTINCT u.id)::int AS total_clinicas,
      SUM(u.fat_meta_mensal) AS fat_meta_total,
      SUM(u.fat_minimo_mensal) AS fat_minimo_total,
      SUM(u.fat_maximo_mensal) AS fat_maximo_total,
      (SELECT COUNT(*) FROM pacientes WHERE ativo = true) AS total_pacientes,
      (SELECT SUM(valor_realizado) FROM faturamento_diario
        WHERE data >= date_trunc('month', CURRENT_DATE)) AS fat_realizado_mes,
      (SELECT SUM(consultas_count) FROM faturamento_diario
        WHERE data = CURRENT_DATE) AS consultas_hoje,
      (SELECT SUM(receitas_fama) FROM faturamento_diario
        WHERE data >= date_trunc('month', CURRENT_DATE)) AS receitas_fama_mes,
      (SELECT SUM(comissao_magistral_estimada) FROM faturamento_diario
        WHERE data >= date_trunc('month', CURRENT_DATE)) AS comissao_total_mes
    FROM unidades u WHERE u.fat_meta_mensal > 0
  `)
  res.json(rows[0])
})

// GET /api/dashboard/clinicas/ranking
// Ranking de clínicas por faturamento — CEO vê quem fatura mais/menos
router.get('/clinicas/ranking', requireMaster, async (req, res) => {
  const { rows } = await pool.query(`
    SELECT
      u.id, u.nome, u.slug,
      u.fat_minimo_mensal, u.fat_maximo_mensal, u.fat_meta_mensal,
      COALESCE(SUM(fd.valor_realizado), 0) AS fat_realizado_mes,
      COALESCE(SUM(fd.consultas_count), 0) AS consultas_mes,
      COALESCE(AVG(fd.ticket_medio), 0) AS ticket_medio,
      COALESCE(SUM(fd.receitas_fama), 0) AS receitas_fama_mes,
      COALESCE(SUM(fd.comissao_magistral_estimada), 0) AS comissao_mes,
      ROUND(
        COALESCE(SUM(fd.valor_realizado), 0) / NULLIF(u.fat_meta_mensal, 0) * 100
      , 1) AS pct_meta
    FROM unidades u
    LEFT JOIN faturamento_diario fd ON fd.unidade_id = u.id
      AND fd.data >= date_trunc('month', CURRENT_DATE)
    WHERE u.fat_meta_mensal > 0
    GROUP BY u.id, u.nome, u.slug, u.fat_minimo_mensal, u.fat_maximo_mensal, u.fat_meta_mensal
    ORDER BY fat_realizado_mes DESC
  `)
  res.json(rows)
})

// src/components/indicators/Gauge.tsx — velocímetro SVG customizado
interface GaugeProps {
  size?: number
  value: number
  min?: number
  max?: number
  target?: number
  label: string
  unit?: string
  color?: string
}

export function Gauge({ size = 164, value, min = 0, max = 100, target, label, unit = '%', color }: GaugeProps) {
  const { getBand, PAWARDS } = require('../../theme/pawards-tokens')
  const pct = Math.max(0, Math.min(1, (value - min) / (max - min)))
  const band = getBand(Math.round(pct * 100))
  const arcColor = color ?? band.color

  const cx = size / 2, cy = size / 2
  const r = size * 0.38
  const strokeW = size * 0.08
  const startAngle = -220, endAngle = 40
  const totalDeg = endAngle - startAngle
  const valueDeg = startAngle + totalDeg * pct

  function polarToXY(deg: number, radius: number) {
    const rad = (deg - 90) * Math.PI / 180
    return { x: cx + radius * Math.cos(rad), y: cy + radius * Math.sin(rad) }
  }

  function arc(start: number, end: number, radius: number) {
    const s = polarToXY(start, radius)
    const e = polarToXY(end, radius)
    const large = (end - start) > 180 ? 1 : 0
    return `M ${s.x} ${s.y} A ${radius} ${radius} 0 ${large} 1 ${e.x} ${e.y}`
  }

  const needleEnd = polarToXY(valueDeg, r * 0.75)

  return (
    <div style={{ textAlign: 'center' }}>
      <svg width={size} height={size} viewBox={`0 0 ${size} ${size}`}>
        {/* Arco de fundo */}
        <path d={arc(startAngle, endAngle, r)} fill="none"
          stroke="rgba(255,255,255,0.06)" strokeWidth={strokeW} strokeLinecap="round" />
        {/* Arco colorido */}
        <path d={arc(startAngle, valueDeg, r)} fill="none"
          stroke={arcColor} strokeWidth={strokeW} strokeLinecap="round"
          style={{ filter: `drop-shadow(0 0 6px ${band.glow})` }} />
        {/* Ponteiro */}
        <line x1={cx} y1={cy} x2={needleEnd.x} y2={needleEnd.y}
          stroke={PAWARDS.colors.text.primary} strokeWidth={1.5} strokeLinecap="round" />
        {/* Hub */}
        <circle cx={cx} cy={cy} r={size * 0.045}
          fill={PAWARDS.colors.bg[800]} stroke={arcColor} strokeWidth={1.5} />
        {/* Valor */}
        <text x={cx} y={cy + size * 0.12} textAnchor="middle"
          fill={PAWARDS.colors.text.primary} fontSize={size * 0.16} fontWeight={700}>
          {typeof value === 'number' && value < 10000 ? value.toFixed(1) : value}
        </text>
        <text x={cx} y={cy + size * 0.24} textAnchor="middle"
          fill={PAWARDS.colors.text.muted} fontSize={size * 0.08}>
          {unit}
        </text>
      </svg>
      <div style={{ fontSize: 11, letterSpacing: '0.08em', fontWeight: 600,
        color: PAWARDS.colors.text.tertiary, textTransform: 'uppercase', marginTop: 4 }}>
        {label}
      </div>
    </div>
  )
}

// src/hooks/useRealtimeDashboard.ts
import { useEffect, useRef, useState } from 'react'

export function useRealtimeDashboard<T>(
  endpoint: string,
  intervalMs = 60_000,
  options?: { enabled?: boolean }
) {
  const [data, setData] = useState<T | null>(null)
  const [loading, setLoading] = useState(true)
  const [error, setError] = useState<string | null>(null)
  const timerRef = useRef<ReturnType<typeof setInterval> | null>(null)

  const fetchData = async () => {
    try {
      const res = await fetch(endpoint, { credentials: 'include' })
      if (!res.ok) throw new Error(`HTTP ${res.status}`)
      const json = await res.json()
      setData(json)
      setError(null)
    } catch (e) {
      setError((e as Error).message)
    } finally {
      setLoading(false)
    }
  }

  useEffect(() => {
    if (options?.enabled === false) return
    fetchData()
    timerRef.current = setInterval(fetchData, intervalMs)
    return () => { if (timerRef.current) clearInterval(timerRef.current) }
  }, [endpoint, intervalMs])

  return { data, loading, error, refresh: fetchData }
}
