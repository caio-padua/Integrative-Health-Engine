No endpoint POST /admin/parmavault/comissao/recalcular, trocar o COALESCE atual por:

WITH calc AS (
  SELECT pr.id,
    COALESCE(
      -- 1o: soma dos valor_maximo dos blocos da prescrição (interno, não exibido)
      NULLIF((
        SELECT SUM(bt.valor_maximo)
        FROM prescricao_blocos pb
        JOIN bloco_template bt ON bt.id = pb.bloco_template_id
        WHERE pb.prescricao_id = pr.prescricao_id
          AND bt.valor_maximo > 0
      ), 0),
      -- 2o: valor declarado pela farmácia
      NULLIF(pr.valor_formula_real, 0),
      -- 3o: estimado
      NULLIF(pr.valor_formula_estimado, 0)
    ) AS base_calculo,
    fp.percentual_comissao
  FROM parmavault_receitas pr
  JOIN farmacias_parmavault fp ON fp.id = pr.farmacia_id
  WHERE pr.comissao_estimada IS NULL OR pr.comissao_estimada = 0
)
UPDATE parmavault_receitas pr SET
  comissao_estimada = ROUND(c.base_calculo * (c.percentual_comissao / 100), 2),
  comissao_estimada_origem = 'valor_maximo_bloco_template'
FROM calc c WHERE c.id = pr.id AND c.base_calculo IS NOT NULL
