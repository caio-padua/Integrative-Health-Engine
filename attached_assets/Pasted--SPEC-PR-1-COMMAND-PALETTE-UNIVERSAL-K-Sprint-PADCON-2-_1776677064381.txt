## 1. OBJETIVO

Implantar o atalho universal ⌘K (Mac) / Ctrl+K (Win/Linux) que abre um painel de busca de comandos acima de qualquer tela. É o **teletransporte verdadeiro** descrito no Manifesto 2.0, princípio #6 — qualquer tela, qualquer ação, três teclas de distância.

Este é o primeiro PR de frontend do ciclo de 4 meses. Ele define o **padrão arquitetural de extensibilidade por módulo** que se repete nos próximos 13 passos. Executar ele bem significa que todo módulo futuro herda um registry de comandos automaticamente. Executar mal significa retrabalho em cada uma das 11 telas da Fase A.

Prioridade: **qualidade arquitetural > quantidade de comandos iniciais**. Melhor entregar 15 comandos impecavelmente estruturados do que 60 frouxos.

## 5. TIPOS TYPESCRIPT

`lib/commandPalette/types.ts`:

```ts
import type { LucideIcon } from 'lucide-react';

export type CommandCategory =
  | 'navegar'
  | 'buscar'
  | 'criar'
  | 'acoes'
  | 'admin';

export interface CommandContext {
  navigate: (path: string) => void;
  closePalette: () => void;
  perfilUsuario?: string; // usado depois do JWT; provisório por enquanto
}

export interface Command {
  id: string;                    // único global, formato 'modulo:acao'
  label: string;                 // texto visível
  description?: string;          // subtítulo opcional
  category: CommandCategory;
  keywords: string[];            // termos extras pra match (cpf, nota, etc)
  icon: LucideIcon;              // componente lucide-react
  shortcut?: string[];           // ex: ['⌘', 'P'] — exibido mas não ativado nesta versão
  perform: (ctx: CommandContext) => void | Promise<void>;
  requiresPerfil?: string[];     // placeholder para pós-JWT
  hidden?: boolean;              // esconde da lista mas ainda registrado
}
```

Regra de nomenclatura para `id`: sempre `modulo:acao`, snake_case.
- `navegar:dashboard`
- `pacientes:buscar`
- `nfe:emitir`
- `admin:gateways`

## 6. REGISTRY API

`lib/commandPalette/registry.ts`:

```ts
import type { Command } from './types';

class CommandRegistry {
  private commands = new Map<string, Command>();
  private listeners = new Set<() => void>();

  register(cmd: Command): void {
    if (this.commands.has(cmd.id)) {
      console.warn(`[CommandPalette] comando duplicado: ${cmd.id}`);
      return;
    }
    this.commands.set(cmd.id, cmd);
    this.notify();
  }

  unregister(id: string): void {
    this.commands.delete(id);
    this.notify();
  }

  list(): Command[] {
    return Array.from(this.commands.values())
      .filter(c => !c.hidden)
      .sort((a, b) => a.label.localeCompare(b.label, 'pt-BR'));
  }

  get(id: string): Command | undefined {
    return this.commands.get(id);
  }

  subscribe(fn: () => void): () => void {
    this.listeners.add(fn);
    return () => this.listeners.delete(fn);
  }

  private notify(): void {
    this.listeners.forEach(fn => fn());
  }
}

export const commandRegistry = new CommandRegistry();
```

**Singleton.** Não instanciar múltiplos. Exportado como valor constante, não factory.

## 7. PROVIDER E HOOKS

`lib/commandPalette/provider.tsx`:

```tsx
import { createContext, useContext, useState, useEffect, useCallback } from 'react';
import { useNavigate } from 'wouter';
import { commandRegistry } from './registry';
import { defaultCommands } from './defaultCommands';
import type { Command, CommandContext } from './types';

interface PaletteState {
  isOpen: boolean;
  open: () => void;
  close: () => void;
  toggle: () => void;
  recentCommandIds: string[];
}

const Ctx = createContext<PaletteState | null>(null);

export function CommandPaletteProvider({ children }: { children: React.ReactNode }) {
  const [isOpen, setIsOpen] = useState(false);
  const [recentCommandIds, setRecentCommandIds] = useState<string[]>(() => {
    try {
      return JSON.parse(localStorage.getItem('padcon:cmdk:recent') || '[]');
    } catch {
      return [];
    }
  });

  // registra comandos default no mount
  useEffect(() => {
    defaultCommands.forEach(cmd => commandRegistry.register(cmd));
    return () => defaultCommands.forEach(cmd => commandRegistry.unregister(cmd.id));
  }, []);

  // atalho global ⌘K / Ctrl+K
  useEffect(() => {
    const onKey = (e: KeyboardEvent) => {
      if ((e.metaKey || e.ctrlKey) && e.key.toLowerCase() === 'k') {
        e.preventDefault();
        setIsOpen(v => !v);
      }
    };
    window.addEventListener('keydown', onKey);
    return () => window.removeEventListener('keydown', onKey);
  }, []);

  const open = useCallback(() => setIsOpen(true), []);
  const close = useCallback(() => setIsOpen(false), []);
  const toggle = useCallback(() => setIsOpen(v => !v), []);

  return (
    <Ctx.Provider value={{ isOpen, open, close, toggle, recentCommandIds }}>
      {children}
    </Ctx.Provider>
  );
}

export function useCommandPalette(): PaletteState {
  const ctx = useContext(Ctx);
  if (!ctx) throw new Error('useCommandPalette fora de CommandPaletteProvider');
  return ctx;
}
```

`lib/commandPalette/hooks.ts`:

```ts
import { useEffect } from 'react';
import { commandRegistry } from './registry';
import type { Command } from './types';

export function useRegisterCommand(cmd: Command | Command[]): void {
  useEffect(() => {
    const arr = Array.isArray(cmd) ? cmd : [cmd];
    arr.forEach(c => commandRegistry.register(c));
    return () => arr.forEach(c => commandRegistry.unregister(c.id));
  }, []); // intencionalmente vazio — comando é estável no ciclo de vida do componente
}
```

## 9. COMPONENTE VISUAL

`components/CommandPalette/CommandPalette.tsx`:

```tsx
import { Command as CmdkRoot } from 'cmdk';
import { useCommandPalette } from '@/lib/commandPalette';
import { useNavigate } from 'wouter';
import { commandRegistry } from '@/lib/commandPalette/registry';
import { useEffect, useState } from 'react';
import { CommandItem } from './CommandItem';
import { CommandGroup } from './CommandGroup';
import { CommandFooter } from './CommandFooter';
import './tokens.css';

export function CommandPalette() {
  const { isOpen, close, recentCommandIds } = useCommandPalette();
  const [, navigate] = useNavigate();
  const [commands, setCommands] = useState(commandRegistry.list());

  useEffect(() => {
    return commandRegistry.subscribe(() => setCommands(commandRegistry.list()));
  }, []);

  if (!isOpen) return null;

  const ctx = { navigate, closePalette: close };
  const grouped = groupByCategory(commands);
  const recents = recentCommandIds
    .map(id => commandRegistry.get(id))
    .filter((c): c is NonNullable<typeof c> => Boolean(c))
    .slice(0, 5);

  return (
    <div
      className="cmdk-overlay"
      onClick={close}
      role="dialog"
      aria-modal="true"
      aria-label="Paleta de comandos"
    >
      <div className="cmdk-container" onClick={(e) => e.stopPropagation()}>
        <CmdkRoot label="Paleta de comandos" loop>
          <CmdkRoot.Input
            className="cmdk-input"
            placeholder="Buscar comandos ou digite para agir..."
            autoFocus
          />
          <CmdkRoot.List className="cmdk-list">
            <CmdkRoot.Empty className="cmdk-empty">
              Nenhum comando encontrado.
            </CmdkRoot.Empty>

            {recents.length > 0 && (
              <CommandGroup heading="Recentes">
                {recents.map(cmd => (
                  <CommandItem key={`recent-${cmd.id}`} command={cmd} context={ctx} />
                ))}
              </CommandGroup>
            )}

            {(['navegar', 'buscar', 'criar', 'acoes', 'admin'] as const).map(cat => {
              const items = grouped[cat] || [];
              if (items.length === 0) return null;
              return (
                <CommandGroup key={cat} heading={categoryLabel(cat)}>
                  {items.map(cmd => (
                    <CommandItem key={cmd.id} command={cmd} context={ctx} />
                  ))}
                </CommandGroup>
              );
            })}
          </CmdkRoot.List>
          <CommandFooter />
        </CmdkRoot>
      </div>
    </div>
  );
}

function groupByCategory(cmds: Command[]) {
  return cmds.reduce<Record<string, Command[]>>((acc, cmd) => {
    (acc[cmd.category] = acc[cmd.category] || []).push(cmd);
    return acc;
  }, {});
}

function categoryLabel(cat: string): string {
  return { navegar: 'Navegar', buscar: 'Buscar', criar: 'Criar',
           acoes: 'Ações', admin: 'Admin' }[cat] || cat;
}
```

## 11. RECENT COMMANDS

Em `provider.tsx`, adicionar função `recordUse(id)` exposta no contexto:

```ts
const recordUse = useCallback((id: string) => {
  setRecentCommandIds(prev => {
    const next = [id, ...prev.filter(x => x !== id)].slice(0, 5);
    localStorage.setItem('padcon:cmdk:recent', JSON.stringify(next));
    return next;
  });
}, []);
```

E no `CommandItem.tsx`, chamar `recordUse(command.id)` antes de `perform(ctx)`.

Guarda apenas os **5 últimos distintos**. Não guarda quando o comando é `hidden`. Limpar via `localStorage.removeItem('padcon:cmdk:recent')` se necessário.
