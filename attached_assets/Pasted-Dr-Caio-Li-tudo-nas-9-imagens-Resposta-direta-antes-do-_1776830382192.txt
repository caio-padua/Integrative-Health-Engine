// requireRole helper
function requireRole(...perfis: string[]) {
  return (req: Request, res: Response, next: NextFunction) => {
    if (!perfis.includes(req.user?.perfil ?? '')) {
      res.status(403).json({ error: 'Acesso negado' });
      return;
    }
    next();
  };
}

// Aplicar em todos os endpoints /admin/analytics:
router.get('/admin/analytics/crescimento-clinicas',
  requireRole('validador_mestre', 'consultoria_master'),
  async (req, res) => {
    const { periodo_a, periodo_b, metrica = 'faturamento' } = req.query;
    const [ano_a, mes_a] = String(periodo_a).split('-').map(Number);
    const [ano_b, mes_b] = String(periodo_b).split('-').map(Number);

    // Switch de métrica — furo 4 corrigido
    const coluna = metrica === 'receitas' ? 'receitas_count'
      : metrica === 'consultas' ? 'consultas_count'
      : 'faturamento_total';

    const result = await db.execute(sql`
      SELECT
        u.id AS unidade_id,
        u.nome,
        u.tipo,
        COALESCE(a.${sql.raw(coluna)}, 0) AS valor_a,
        COALESCE(b.${sql.raw(coluna)}, 0) AS valor_b,
        CASE
          WHEN COALESCE(a.${sql.raw(coluna)}, 0) = 0 THEN NULL
          ELSE ROUND(
            ((COALESCE(b.${sql.raw(coluna)}, 0) - COALESCE(a.${sql.raw(coluna)}, 0))
             / COALESCE(a.${sql.raw(coluna)}, 1)::float) * 100, 1
          )
        END AS variacao_pct,
        COALESCE(b.${sql.raw(coluna)}, 0) - COALESCE(a.${sql.raw(coluna)}, 0) AS variacao_absoluta
      FROM unidades u
      LEFT JOIN analytics_clinica_mes a
        ON a.unidade_id = u.id AND a.ano = ${ano_a} AND a.mes = ${mes_a}
      LEFT JOIN analytics_clinica_mes b
        ON b.unidade_id = u.id AND b.ano = ${ano_b} AND b.mes = ${mes_b}
      WHERE u.ativa = true
      ORDER BY variacao_pct DESC NULLS LAST
    `);

    res.json(result.rows.map((r: any) => ({
      ...r,
      tendencia: r.variacao_pct > 5 ? 'CRESCENDO'
        : r.variacao_pct < -5 ? 'CAINDO' : 'ESTAVEL'
    })));
  }
);
