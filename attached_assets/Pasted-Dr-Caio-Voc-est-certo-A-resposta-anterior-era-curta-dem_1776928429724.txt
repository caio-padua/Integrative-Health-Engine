// Substituir o parser manual na rota POST /declaracoes/csv
// por parsing linha a linha com suporte a aspas:

function parseCsvLinha(linha: string): string[] {
  const cols: string[] = [];
  let atual = "";
  let dentroAspas = false;
  for (let i = 0; i < linha.length; i++) {
    const c = linha[i];
    if (c === '"') {
      dentroAspas = !dentroAspas;
    } else if (c === "," && !dentroAspas) {
      cols.push(atual.trim());
      atual = "";
    } else {
      atual += c;
    }
  }
  cols.push(atual.trim());
  return cols;
}

if (["FAMA", "HORM", "FITO"].includes(destino_dispensacao)) {

      LIMIT 1
